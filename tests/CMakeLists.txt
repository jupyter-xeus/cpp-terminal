if(CMAKE_CXX_COMPILER_ID STREQUAL "GNU" AND CMAKE_CXX_COMPILER_VERSION VERSION_GREATER_EQUAL 4.7.0 AND CMAKE_CXX_COMPILER_VERSION VERSION_LESS 4.7.9)
  return()
endif()

if(CMAKE_VERSION GREATER_EQUAL 3.11)
  include(FetchContent)
  FetchContent_Declare(doctest GIT_REPOSITORY https://github.com/doctest/doctest.git GIT_TAG v2.4.9)
  FetchContent_GetProperties(doctest)
  if(NOT doctest_POPULATED)
    FetchContent_Populate(doctest)
    add_subdirectory("${doctest_SOURCE_DIR}" "${doctest_BINARY_DIR}")
    include("${doctest_SOURCE_DIR}/scripts/cmake/doctest.cmake")
  endif()
else()
  find_package(doctest QUIET)
  if(NOT doctest_FOUND)
    message(WARNING "CMake is too old to use FetchContent and doctest is not found !\nSkipping tests !")
    return()
  endif()
  include(doctest)
endif()

add_executable(test_terminal test_terminal.cpp)
target_link_libraries(test_terminal PRIVATE doctest::doctest cpp-terminal::cpp-terminal)
doctest_discover_tests(test_terminal)

add_test(NAME colors COMMAND "${CMAKE_BINARY_DIR}/examples/colors${CMAKE_EXECUTABLE_SUFFIX_CXX}")

if(WIN32)
  find_program(POWERSHELL_EXECUTABLE NAMES powershell DOC "PowerShell command")
  add_test(NAME read_stdin_not_tty COMMAND ${POWERSHELL_EXECUTABLE} echo 'Hello world' | "${CMAKE_BINARY_DIR}/examples/read_stdin${CMAKE_EXECUTABLE_SUFFIX_CXX}")
  add_test(NAME colors_not_tty COMMAND ${POWERSHELL_EXECUTABLE} echo 'Hello world' | "${CMAKE_BINARY_DIR}/examples/colors${CMAKE_EXECUTABLE_SUFFIX_CXX}")
  add_test(NAME keys_not_tty COMMAND "${CMAKE_BINARY_DIR}/examples/keys${CMAKE_EXECUTABLE_SUFFIX_CXX}")
endif()
set_tests_properties(read_stdin_not_tty PROPERTIES PASS_REGULAR_EXPRESSION "Input from stdin: Hello world")
set_tests_properties(keys_not_tty PROPERTIES WILL_FAIL TRUE)
